            function
            sendWeeklyLessonPlans() {
              const
            ss = SpreadsheetApp.getActiveSpreadsheet();
              const
            teachersSheet = ss.getSheetByName(&quot;Teachers&quot;);
              const
            lessonsSheet = ss.getSheetByName(&quot;Lesson Plans&quot;);


              const
            teachers = teachersSheet.getDataRange().getValues();
              const
            lessons = lessonsSheet.getDataRange().getValues();


              const
            today = new Date();
              const
            nextWeek = new Date(today);

            nextWeek.setDate(today.getDate()
            + 7);


              for
            (let i = 1; i &lt; teachers.length; i++) {
                const
            teacherEmail = teachers[i][2];
                const
            teacherID = teachers[i][0];


                //
            Filter lessons for this teacher in the next week
                let
            upcomingLessons = lessons.filter(row =&gt; {
                  const
            lessonDate = new Date(row[1]);
                  return
            row[2] === teacherID &amp;&amp; lessonDate &gt;= today &amp;&amp;
            lessonDate &lt;= nextWeek;
                });


                if
            (upcomingLessons.length &gt; 0) {
                  let
            message = &quot;Hello &quot; + teachers[i][1] + &quot;,\n\nHere are
            your lessons for the upcoming week:\n\n&quot;;

            upcomingLessons.forEach(lesson
            =&gt; {
                    message
            += `- ${lesson[1]}: ${lesson[3]} - ${lesson[4]}\n`;
                  });
                  message
            += &quot;\nBest regards,\nTeachAI&quot;;



            MailApp.sendEmail(teacherEmail,
            &quot;Your Weekly Lesson Plan Summary&quot;, message);
                }
              }
            }
